    **1. Add `cmdRequirementUpdateStatus` function to `lib/commands.js`:**

    Add after `cmdRequirementTraceability` in the Requirements section:

    ```javascript
    /** Valid requirement statuses */
    const VALID_REQUIREMENT_STATUSES = ['Pending', 'In Progress', 'Done', 'Deferred'];

    /**
     * CLI command: Update the status of a requirement in the Traceability Matrix.
     * Validates REQ-ID exists and status is valid before writing.
     * @param {string} cwd - Project working directory
     * @param {string} reqId - Requirement ID (e.g., "REQ-31")
     * @param {string} newStatus - New status value (Pending, In Progress, Done, Deferred)
     * @param {boolean} raw - Output raw JSON string instead of pretty-printed
     */
    function cmdRequirementUpdateStatus(cwd, reqId, newStatus, raw) {
      // Validate status
      if (!VALID_REQUIREMENT_STATUSES.includes(newStatus)) {
        error(`Invalid status "${newStatus}". Valid statuses: ${VALID_REQUIREMENT_STATUSES.join(', ')}`);
      }

      const reqFilePath = path.join(cwd, '.planning', 'REQUIREMENTS.md');
      const content = safeReadFile(reqFilePath);
      if (!content) {
        error('REQUIREMENTS.md not found');
      }

      // Parse traceability matrix to validate REQ-ID exists and get old status
      const matrix = parseTraceabilityMatrix(content);
      const matrixRow = matrix.find(
        (m) => m.req.toLowerCase() === reqId.toLowerCase()
      );
      if (!matrixRow) {
        error(`Requirement ${reqId} not found in Traceability Matrix`);
      }

      const oldStatus = matrixRow.status;

      // Update the traceability matrix row in the raw content
      // Strategy: find the table row matching this REQ-ID and replace the status cell
      const lines = content.split('\n');
      let updated = false;
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];
        // Match table rows that start with | and contain the REQ-ID
        if (line.startsWith('|') && line.toLowerCase().includes(reqId.toLowerCase())) {
          const cells = line.split('|').map((c) => c.trim());
          // cells[0] is empty (before first |), cells[1]=REQ, cells[2]=Feature, cells[3]=Priority, cells[4]=Phase, cells[5]=Status
          if (cells.length >= 6 && cells[1].toLowerCase() === reqId.toLowerCase()) {
            cells[5] = ` ${newStatus} `;
            // Reconstruct row preserving pipe structure
            // Skip empty first/last elements from split
            const reconstructed = '| ' + cells.slice(1, -1).map((c, idx) => {
              if (idx === 4) return newStatus; // status column (index 4 in slice = cells[5])
              return c;
            }).join(' | ') + ' |';
            lines[i] = reconstructed;
            updated = true;
            break;
          }
        }
      }

      if (!updated) {
        error(`Failed to update status for ${reqId} in Traceability Matrix`);
      }

      // Write updated content back to disk
      fs.writeFileSync(reqFilePath, lines.join('\n'), 'utf-8');

      output({
        updated: true,
        id: matrixRow.req, // Use original casing from file
        old_status: oldStatus,
        new_status: newStatus,
      }, raw);
    }
    ```

    **Important implementation detail for the table row update:**

    The traceability matrix table has this format:
    ```
    | REQ-01 | Core Feature | P0 | Phase 1 | Done |
    ```

    When splitting by `|`, we get: `['', ' REQ-01 ', ' Core Feature ', ' P0 ', ' Phase 1 ', ' Done ', '']`

    The status is at index 5 (0-indexed). We need to:
    1. Split the line by `|`
    2. Find the row where the trimmed cell[1] matches the REQ-ID (case-insensitive)
    3. Replace cell[5] with the new status
    4. Rejoin with `|` preserving the original spacing style

    A cleaner approach for the row replacement that preserves formatting:
    ```javascript
    // Use regex to replace the last cell content before the trailing |
    // Pattern: | REQ-ID | ... | old_status |
    // Replace old_status with new_status
    const reqPattern = new RegExp(
      `^(\\|\\s*${reqId.replace('-', '\\-')}\\s*\\|.+\\|)\\s*${oldStatus.replace(' ', '\\s+')}\\s*(\\|)\\s*$`,
      'im'
    );
    const updatedContent = content.replace(reqPattern, `$1 ${newStatus} $2`);
    ```

    Use whichever approach is cleaner. The key constraint is:
    - Only the status cell changes
    - All other cells and content are preserved exactly
    - The file is written back with the same line endings
