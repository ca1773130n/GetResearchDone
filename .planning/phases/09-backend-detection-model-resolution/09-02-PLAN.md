    function resolveModelInternal(cwd, agentType) {
      const config = loadConfig(cwd);
      const profile = config.model_profile || 'balanced';
      const agentModels = MODEL_PROFILES[agentType];

    New implementation:
    ```javascript
    function resolveModelInternal(cwd, agentType) {
      const config = loadConfig(cwd);
      const profile = config.model_profile || 'balanced';
      const agentModels = MODEL_PROFILES[agentType];
      if (!agentModels) {
        // Unknown agent type: resolve 'sonnet' tier through backend
        const backend = detectBackend(cwd);
        return resolveBackendModel(backend, 'sonnet', config);
      }
      const tier = agentModels[profile] || agentModels['balanced'] || 'sonnet';
      const backend = detectBackend(cwd);
      return resolveBackendModel(backend, tier, config);
    }
    ```

    **Modification 4: Update resolveModelForAgent() similarly**

    Current:
    ```javascript
    function resolveModelForAgent(config, agentType) {
      const profile = (config.model_profile || 'balanced').toLowerCase();
      const agentModels = MODEL_PROFILES[agentType];
      if (!agentModels) return 'sonnet';
      return agentModels[profile] || agentModels['balanced'] || 'sonnet';
    }
    ```

    New -- note this function takes config directly, not cwd, so it needs a backend parameter or needs to detect. Since it's called from context.js which already has cwd, add backend detection:
    ```javascript
    function resolveModelForAgent(config, agentType, cwd) {
      const profile = (config.model_profile || 'balanced').toLowerCase();
      const agentModels = MODEL_PROFILES[agentType];
      const tier = agentModels
        ? (agentModels[profile] || agentModels['balanced'] || 'sonnet')
        : 'sonnet';
      // If cwd provided, resolve to backend-specific model name
      if (cwd) {
        const backend = detectBackend(cwd);
        return resolveBackendModel(backend, tier, config);
      }
      // Backward compatible: no cwd means return tier name (existing behavior)
      return tier;
    }
    ```

    This preserves backward compatibility -- existing callers that don't pass `cwd` get the old behavior (tier names). New callers can pass `cwd` to get backend-specific names.

    **IMPORTANT:** Do NOT change any function signatures in a breaking way. The `cwd` parameter on `resolveModelForAgent` is optional (added at the end).
