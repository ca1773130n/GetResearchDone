  <name>Task 2: Wire CLI command and MCP tool for markdown-split</name>
  <files>bin/grd-tools.js, lib/mcp-server.js, lib/commands.js</files>
  <action>
    **Step 1: Add CLI command wrapper in `lib/commands.js`**

    Import `splitMarkdown` from `./markdown-split`:
    ```javascript
    const { splitMarkdown } = require('./markdown-split');
    ```

    Add command function:
    ```javascript
    function cmdMarkdownSplit(cwd, filePath, args, raw) {
      if (!filePath) error('file path required for markdown-split');

      const absPath = path.isAbsolute(filePath) ? filePath : path.join(cwd, filePath);
      if (!fs.existsSync(absPath)) error(`File not found: ${filePath}`);

      const threshold = undefined; // use default
      const thresholdIdx = args.indexOf('--threshold');
      const thresholdVal = thresholdIdx !== -1 ? parseInt(args[thresholdIdx + 1], 10) : undefined;

      const result = splitMarkdown(absPath, thresholdVal ? { threshold: thresholdVal } : {});
      output(result, raw, result.split ? `Split into ${result.partials.length} partials` : 'No split needed');
    }
    ```

    Export `cmdMarkdownSplit` from the module.

    **Step 2: Wire in `bin/grd-tools.js`**

    Import `cmdMarkdownSplit` from `../lib/commands`:
    ```javascript
    const { ..., cmdMarkdownSplit } = require('../lib/commands');
    ```

    Add case in the routeCommand switch:
    ```javascript
    case 'markdown-split': {
      const file = args[0];
      cmdMarkdownSplit(cwd, file, args.slice(1), raw);
      break;
    }
    ```

    Add to the usage string in the help text section.

    **Step 3: Add MCP tool descriptor in `lib/mcp-server.js`**

    Add to the COMMAND_DESCRIPTORS array:
    ```javascript
    {
      name: 'markdown_split',
      description: 'Split a large markdown file into indexed partials. Files exceeding ~25,000 tokens are split at heading boundaries into numbered partial files. The original file becomes an index linking to partials. Idempotent: re-running on already-split files is a no-op.',
      params: {
        file_path: { type: 'string', description: 'Path to the markdown file to split (relative to project root)', required: true },
        threshold: { type: 'number', description: 'Token threshold for splitting (default: 25000)', required: false },
      },
      execute: (cwd, params) => {
        const { splitMarkdown } = require('./markdown-split');
        const absPath = path.isAbsolute(params.file_path)
          ? params.file_path
          : path.join(cwd, params.file_path);
        return splitMarkdown(absPath, params.threshold ? { threshold: params.threshold } : {});
      },
    }
    ```

    Import path at top if not already imported.
