    ```javascript
    /**
     * Push worktree branch to remote and create a PR via gh CLI.
     *
     * @param {string} cwd - Project working directory (main repo, NOT worktree)
     * @param {Object} options
     * @param {string} options.phase - Phase number (required)
     * @param {string} [options.milestone] - Milestone version (defaults from ROADMAP.md)
     * @param {string} [options.title] - PR title (defaults to "Phase {phase}: {slug}")
     * @param {string} [options.body] - PR body markdown
     * @param {string} [options.base] - Base branch for PR target (defaults from config)
     * @param {boolean} raw - Output raw text instead of JSON
     */
    function cmdWorktreePushAndPR(cwd, options, raw) {
      // 1. Validate phase
      // 2. Resolve milestone, compute worktree path and branch
      // 3. Verify worktree directory exists on disk
      // 4. Push branch to origin from worktree directory:
      //    execGit(wtPath, ['push', '-u', 'origin', branch], { allowBlocked: true })
      //    Note: push from wtPath (worktree cwd), not main repo cwd
      // 5. Resolve base branch: options.base || loadConfig(cwd).base_branch || 'main'
      // 6. Build PR title: options.title || `Phase ${phase}: ${slug} (${milestone})`
      // 7. Build PR body: options.body || default body with phase/milestone info
      // 8. Create PR via gh CLI:
      //    execFileSync('gh', ['pr', 'create',
      //      '--title', title,
      //      '--body', body,
      //      '--base', baseBranch,
      //      '--head', branch
      //    ], { cwd: wtPath, stdio: 'pipe', encoding: 'utf-8' })
      // 9. Parse gh output for PR URL (gh pr create outputs the URL on stdout)
      // 10. Return structured JSON: { pr_url, branch, title, base, phase, milestone }
      //
      // Error handling:
      // - Push failure: return { error: "Failed to push branch", details: stderr }
      // - gh failure: return { error: "Failed to create PR", details: stderr }
      //   Include push_succeeded: true so caller knows branch was pushed even if PR failed
      // - Both use output() not error() (exit 0, not exit 1) so callers get JSON
    }
    ```

    - Use `execFileSync('gh', ...)` directly for gh CLI (not through execGit since gh is not git)
    - Wrap gh call in try/catch, return structured error on failure
    - The PR number can be extracted from the URL: `url.split('/').pop()`
    - Add `cmdWorktreePushAndPR` to the module.exports
