          echo "grd-tools CLI: OK"

          # Verify plugin.json exists at expected path
          test -f node_modules/grd-tools/.claude-plugin/plugin.json
          echo "plugin.json: OK"

          # Clean up
          cd "$GITHUB_WORKSPACE"
          rm -rf /tmp/grd-test-install
          rm -f "$TARBALL"
