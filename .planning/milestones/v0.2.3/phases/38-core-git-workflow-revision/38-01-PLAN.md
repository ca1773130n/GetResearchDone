       function ensureGitignoreEntry(cwd, entry) {
         const gitignorePath = path.join(cwd, '.gitignore');
         let content = '';
         try {
           content = fs.readFileSync(gitignorePath, 'utf-8');
         } catch {
           // .gitignore doesn't exist yet -- will create it
         }
         // Check if entry already present (exact line match)
         const lines = content.split('\n');
         const normalizedEntry = entry.replace(/\/$/, '');
         const alreadyPresent = lines.some(line => {
           const trimmed = line.trim().replace(/\/$/, '');
           return trimmed === normalizedEntry || trimmed === normalizedEntry + '/';
         });
         if (alreadyPresent) return false;
         // Append entry with a newline
         const suffix = content.endsWith('\n') || content === '' ? '' : '\n';
         fs.writeFileSync(gitignorePath, content + suffix + entry + '\n', 'utf-8');
         return true;
       }
       ```

       function createMilestoneBranch(cwd, milestone, slug) {
         const config = loadConfig(cwd);
         const baseBranch = config.base_branch || 'main';
         const template = config.milestone_branch_template || 'grd/{milestone}-{slug}';
         const branchName = template
           .replace('{milestone}', milestone)
           .replace('{slug}', slug || milestone);

         // Verify current HEAD is on base_branch
         const headResult = execGit(cwd, ['rev-parse', '--abbrev-ref', 'HEAD']);
         if (headResult.exitCode !== 0) {
           return { error: 'Failed to determine current branch', details: headResult.stderr };
         }
         const currentBranch = headResult.stdout.trim();
         if (currentBranch !== baseBranch) {
           return {
             error: `Must be on ${baseBranch} to create milestone branch (currently on ${currentBranch})`,
             current_branch: currentBranch,
             base_branch: baseBranch
           };
         }

         // Check if milestone branch already exists
         const checkResult = execGit(cwd, ['rev-parse', '--verify', branchName]);
         if (checkResult.exitCode === 0) {
           return { branch: branchName, already_exists: true };
         }

         // Create the milestone branch
         const createResult = execGit(cwd, ['branch', branchName]);
         if (createResult.exitCode !== 0) {
           return { error: 'Failed to create milestone branch', details: createResult.stderr };
         }

         return { branch: branchName, created: true, base: baseBranch };
       }
       ```

    2. **Add `resolveTargetBranch(cwd, milestone, slug)` function:**
       This determines the PR target branch based on branching strategy:
       ```js
       function resolveTargetBranch(cwd, milestone, slug) {
         const config = loadConfig(cwd);
         const strategy = config.branching_strategy || 'none';

         if (strategy === 'milestone') {
           // Phase PRs target the milestone branch
           const template = config.milestone_branch_template || 'grd/{milestone}-{slug}';
           return template
             .replace('{milestone}', milestone)
             .replace('{slug}', slug || milestone);
         }

         // For 'phase' strategy (and fallback): PRs target base_branch directly
         return config.base_branch || 'main';
       }
       ```
