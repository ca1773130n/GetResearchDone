       ```js
       /**
        * Run the test suite in a worktree directory as a gate before merge/PR.
        * @param {string} wtPath - Absolute path to the worktree directory
        * @param {string} [testCmd='npm test'] - Test command to execute
        * @returns {{ passed: boolean, exitCode: number, stdout: string, stderr: string }}
        */
       function runTestGate(wtPath, testCmd) {
         const cmd = testCmd || 'npm test';
         const parts = cmd.split(/\s+/);
         const bin = parts[0];
         const args = parts.slice(1);

         try {
           const stdout = execFileSync(bin, args, {
             cwd: wtPath,
             encoding: 'utf-8',
             stdio: 'pipe',
             timeout: 300000, // 5 minute timeout
           });
           return { passed: true, exitCode: 0, stdout: stdout || '', stderr: '' };
         } catch (err) {
           return {
             passed: false,
             exitCode: err.status ?? 1,
             stdout: err.stdout ? err.stdout.toString() : '',
             stderr: err.stderr ? err.stderr.toString() : err.message || 'Test execution failed',
           };
         }
       }
       ```

    2. **Add `cleanupWorktree(cwd, wtPath)` function** -- a finally-block-safe cleanup that never throws:
       ```js
       /**
        * Remove a worktree directory and prune git state. Safe to call in finally blocks.
        * Never throws -- errors are silently absorbed since this is cleanup code.
        * @param {string} cwd - Project working directory (main repo)
        * @param {string} wtPath - Absolute path to the worktree directory
        * @returns {{ cleaned: boolean, error: string|null }}
        */
       function cleanupWorktree(cwd, wtPath) {
         try {
           // Try git worktree remove first (cleans git state + directory)
           if (fs.existsSync(wtPath)) {
             execGit(cwd, ['worktree', 'remove', wtPath, '--force'], { allowBlocked: true });
           }
           // Prune to clean any remaining stale references
           execGit(cwd, ['worktree', 'prune']);
           // If directory still exists after git worktree remove, force-remove it
           if (fs.existsSync(wtPath)) {
             fs.rmSync(wtPath, { recursive: true, force: true });
           }
           return { cleaned: true, error: null };
         } catch (err) {
           // Never throw from cleanup -- absorb errors
           return { cleaned: false, error: err.message || 'Cleanup failed' };
         }
       }
       ```

    1. **Add `mergeWorktree(cwd, options)` function:**
       ```js
       /**
        * Merge a worktree branch into the base branch locally.
        * Does NOT clean up the worktree -- caller should use cleanupWorktree after.
        *
        * @param {string} cwd - Project working directory (main repo, NOT worktree)
        * @param {Object} options
        * @param {string} options.branch - Branch name to merge
        * @param {string} options.baseBranch - Target branch to merge into (e.g., 'main')
        * @returns {{ merged: boolean, branch: string, base: string, error?: string, details?: string }}
        */
       function mergeWorktree(cwd, options) {
         const { branch, baseBranch } = options;
         if (!branch || !baseBranch) {
           return { merged: false, error: 'branch and baseBranch are required' };
         }

         // Checkout the base branch
         const checkoutResult = execGit(cwd, ['checkout', baseBranch]);
         if (checkoutResult.exitCode !== 0) {
           return {
             merged: false,
             branch,
             base: baseBranch,
             error: 'Failed to checkout base branch',
             details: checkoutResult.stderr,
           };
         }

         // Merge the worktree branch
         const mergeResult = execGit(cwd, ['merge', branch, '--no-edit']);
         if (mergeResult.exitCode !== 0) {
           // Abort the failed merge to leave repo in clean state
           execGit(cwd, ['merge', '--abort']);
           return {
             merged: false,
             branch,
             base: baseBranch,
             error: 'Merge failed (conflict or error)',
             details: mergeResult.stderr,
           };
         }

         return { merged: true, branch, base: baseBranch };
       }
       ```

    2. **Add `discardWorktree(cwd, wtPath, branch)` function:**
       ```js
       /**
        * Discard a worktree's work: clean up the worktree and delete the local branch.
        * Uses cleanupWorktree internally for the finally-block-safe removal.
        *
        * @param {string} cwd - Project working directory (main repo)
        * @param {string} wtPath - Absolute path to the worktree directory
        * @param {string} branch - Branch name to delete
        * @returns {{ discarded: boolean, path: string, branch: string, branch_deleted: boolean, error?: string }}
        */
       function discardWorktree(cwd, wtPath, branch) {
         // Clean up the worktree first (removes directory + prunes git state)
         const cleanup = cleanupWorktree(cwd, wtPath);

         // Delete the local branch (use -D for force delete since it may not be merged)
         let branchDeleted = false;
         if (branch) {
           const deleteResult = execGit(cwd, ['branch', '-D', branch], { allowBlocked: true });
           branchDeleted = deleteResult.exitCode === 0;
         }

         return {
           discarded: true,
           path: wtPath,
           branch: branch || '',
           branch_deleted: branchDeleted,
           cleanup_error: cleanup.error,
         };
       }
       ```

    3. **Add `keepWorktree(wtPath)` function:**
       ```js
       /**
        * Keep a worktree intact (no-op). Returns confirmation that the worktree was left in place.
        *
        * @param {string} wtPath - Absolute path to the worktree directory
        * @returns {{ kept: boolean, path: string, exists: boolean }}
        */
       function keepWorktree(wtPath) {
         return {
           kept: true,
           path: wtPath,
           exists: fs.existsSync(wtPath),
         };
       }
       ```
