    1. **Add `cmdWorktreeComplete(cwd, options, raw)` function** in the CLI Commands section, after `cmdWorktreePushAndPR`:

       ```js
       /**
        * Complete a phase worktree with one of 4 actions: merge, pr, keep, discard.
        *
        * - merge: Run test gate, merge branch into base locally, clean up worktree
        * - pr: Run test gate, push branch and create PR, clean up worktree
        * - keep: Leave worktree and branch intact (no test gate, no cleanup)
        * - discard: Remove worktree and delete branch (no test gate)
        *
        * For merge/pr/discard, cleanup runs in a finally block to ensure it
        * happens even if an error occurs mid-operation.
        *
        * @param {string} cwd - Project working directory (main repo, NOT worktree)
        * @param {Object} options
        * @param {string} options.action - One of: 'merge', 'pr', 'keep', 'discard'
        * @param {string} options.phase - Phase number (required)
        * @param {string} [options.milestone] - Milestone version (defaults from ROADMAP.md)
        * @param {string} [options.testCmd] - Test command for the gate (default: 'npm test')
        * @param {string} [options.base] - Base branch override for merge/PR
        * @param {string} [options.title] - PR title (for 'pr' action)
        * @param {string} [options.body] - PR body (for 'pr' action)
        * @param {boolean} raw - If true, output raw text instead of JSON
        */
       function cmdWorktreeComplete(cwd, options, raw) {
         const { action, phase } = options;

         if (!action || !['merge', 'pr', 'keep', 'discard'].includes(action)) {
           output(
             { error: 'action is required and must be one of: merge, pr, keep, discard' },
             raw
           );
           return;
         }

         if (!phase) {
           output({ error: 'phase is required for worktree complete' }, raw);
           return;
         }

         // Resolve milestone and paths
         const milestone = options.milestone || getMilestoneInfo(cwd).version;
         const config = loadConfig(cwd);
         const wtDir = config.worktree_dir || '.worktrees/';
         const wtPath = path.resolve(cwd, wtDir, `${milestone}-${phase}`);

         // Verify worktree exists (except for keep, which can confirm non-existence)
         if (action !== 'keep' && !fs.existsSync(wtPath)) {
           output(
             { error: `Worktree not found at ${wtPath}`, phase, milestone, action },
             raw
           );
           return;
         }

         // Read the branch name from the worktree HEAD
         let branch = null;
         if (fs.existsSync(wtPath)) {
           const headResult = execGit(wtPath, ['rev-parse', '--abbrev-ref', 'HEAD']);
           branch = headResult.exitCode === 0 ? headResult.stdout.trim() : null;
         }

         // Resolve base branch
         const baseBranch = options.base || config.base_branch || 'main';

         // ── KEEP: no test gate, no cleanup ──
         if (action === 'keep') {
           const result = keepWorktree(wtPath);
           output({ action: 'keep', phase, milestone, ...result }, raw);
           return;
         }

         // ── DISCARD: no test gate, cleanup via discardWorktree ──
         if (action === 'discard') {
           const result = discardWorktree(cwd, wtPath, branch);
           output({ action: 'discard', phase, milestone, ...result }, raw);
           return;
         }

         // ── MERGE or PR: test gate required ──
         // Run the test gate first
         const testResult = runTestGate(wtPath, options.testCmd);
         if (!testResult.passed) {
           output(
             {
               action,
               phase,
               milestone,
               blocked: true,
               reason: 'Test gate failed',
               test_exit_code: testResult.exitCode,
               test_stdout: testResult.stdout,
               test_stderr: testResult.stderr,
             },
             raw
           );
           return;
         }

         // Tests passed -- proceed with action, cleanup in finally block
         let actionResult;
         try {
           if (action === 'merge') {
             // Remove worktree BEFORE merge (git won't merge while worktree holds the branch)
             cleanupWorktree(cwd, wtPath);
             actionResult = mergeWorktree(cwd, { branch, baseBranch });

             if (!actionResult.merged) {
               output(
                 {
                   action: 'merge',
                   phase,
                   milestone,
                   error: actionResult.error,
                   details: actionResult.details,
                   test_passed: true,
                 },
                 raw
               );
               return;
             }

             // Delete the branch after successful merge
             execGit(cwd, ['branch', '-d', branch]);

             output(
               {
                 action: 'merge',
                 phase,
                 milestone,
                 merged: true,
                 branch,
                 base: baseBranch,
                 test_passed: true,
               },
               raw
             );

           } else if (action === 'pr') {
             // Push and create PR (operates on the worktree)
             // We capture the push-pr result instead of letting it call output()
             // by calling the helper functions directly

             // Push branch from worktree
             const pushResult = execGit(wtPath, ['push', '-u', 'origin', branch], { allowBlocked: true });
             if (pushResult.exitCode !== 0) {
               output(
                 {
                   action: 'pr',
                   phase,
                   milestone,
                   error: 'Failed to push branch',
                   details: pushResult.stderr,
                   branch,
                   test_passed: true,
                 },
                 raw
               );
               return;
             }

             // Create PR via gh CLI
             const prBase = options.base || baseBranch;
             const phaseInfo = findPhaseInternal(cwd, phase);
             const slug = phaseInfo && phaseInfo.phase_slug
               ? phaseInfo.phase_slug
               : branch.split('/').pop().replace(/^\d+-/, '') || phase;
             const prTitle = options.title || `Phase ${phase}: ${slug} (${milestone})`;
             const prBody = options.body || `## Phase ${phase}\n\nMilestone: ${milestone}\nBranch: ${branch}\n`;

             let prUrl = null;
             let prNumber = null;
             try {
               const ghOutput = execFileSync(
                 'gh',
                 ['pr', 'create', '--title', prTitle, '--body', prBody, '--base', prBase, '--head', branch],
                 { cwd: wtPath, stdio: 'pipe', encoding: 'utf-8' }
               );
               prUrl = ghOutput.trim();
               const urlParts = prUrl.split('/');
               prNumber = urlParts.length > 0 ? urlParts[urlParts.length - 1] : null;
             } catch (ghErr) {
               const stderr = ghErr.stderr
                 ? ghErr.stderr.toString().trim()
                 : ghErr.message || 'Unknown gh error';
               output(
                 {
                   action: 'pr',
                   phase,
                   milestone,
                   error: 'Failed to create PR via gh CLI',
                   details: stderr,
                   push_succeeded: true,
                   branch,
                   base: prBase,
                   title: prTitle,
                   test_passed: true,
                 },
                 raw
               );
               return;
             }

             output(
               {
                 action: 'pr',
                 phase,
                 milestone,
                 pr_url: prUrl,
                 pr_number: prNumber,
                 branch,
                 base: prBase,
                 title: prTitle,
                 test_passed: true,
               },
               raw
             );
           }
         } finally {
           // Cleanup for pr path (merge already cleaned up before merging)
           if (action === 'pr') {
             cleanupWorktree(cwd, wtPath);
           }
         }
       }
       ```

    **Design notes:**
    - For the `merge` action, worktree cleanup happens BEFORE the merge because git refuses to merge a branch that has an active worktree. After the merge succeeds, the branch is deleted with `-d` (lowercase, since it IS merged now).
    - For the `pr` action, the push and PR creation happen while the worktree still exists (the push uses the worktree as cwd). Cleanup happens in the `finally` block afterward.
    - The `finally` block only runs cleanup for the `pr` path because `merge` already removed the worktree before merging, and `discard`/`keep` are handled in their own early-return paths.
    - `cmdWorktreeComplete` calls `output()` (which calls `process.exit`), so the `finally` block still runs since `process.exit` is mocked in tests and `output()` throws through the mock.
    - The PR creation logic is duplicated from `cmdWorktreePushAndPR` rather than calling it, because `cmdWorktreePushAndPR` calls `output()` internally which would trigger process.exit. We need to capture the result and add test_passed/action fields before calling output.
