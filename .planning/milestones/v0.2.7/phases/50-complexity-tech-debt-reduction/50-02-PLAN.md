## Task 1: Add safeReadJSON to lib/utils.js

Add a new exported utility function that combines file reading + JSON parsing with safe fallback:

```js
/**
 * Read and parse a JSON file, returning the default value on any error.
 * @param {string} filePath - Absolute path to the JSON file
 * @param {*} [defaultValue=null] - Value to return if read or parse fails
 * @returns {*} Parsed JSON object, or defaultValue on error
 */
function safeReadJSON(filePath, defaultValue = null) {
  try {
    const raw = fs.readFileSync(filePath, 'utf-8');
    return JSON.parse(raw);
  } catch {
    return defaultValue;
  }
}
```

Add to module.exports.

## Task 2: Add extractMarkdownSection to lib/utils.js

Add a shared helper for the repeated markdown section extraction pattern:

```js
/**
 * Extract content under a markdown heading (## or ### level).
 * @param {string} content - Full markdown content
 * @param {string} heading - Heading text to find (case-insensitive)
 * @param {number} [level=2] - Heading level (2 for ##, 3 for ###)
 * @returns {string|null} Section content (without the heading line), or null if not found
 */
function extractMarkdownSection(content, heading, level = 2) {
  const prefix = '#'.repeat(level);
  const regex = new RegExp(
    `${prefix}\\s+${heading.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')}\\s*\\n([\\s\\S]*?)(?=\\n${prefix}\\s|$)`,
    'i'
  );
  const match = content.match(regex);
  return match ? match[1] : null;
}
```

Add to module.exports.
