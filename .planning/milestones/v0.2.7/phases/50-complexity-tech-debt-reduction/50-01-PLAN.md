## Task 4: Decompose cmdTracker into handler functions

In `lib/tracker.js`, the `cmdTracker` function (starting ~line 411) is a 634-line switch statement with 12 cases: `get-config`, `sync-roadmap`, `sync-phase`, `update-status`, `add-comment`, `sync-status`, `prepare-roadmap-sync`, `prepare-phase-sync`, `record-mapping`, `record-status`, `schedule`, `prepare-reschedule`.

Extract each case body into a named handler function:

```js
function handleGetConfig(cwd, args, raw) { ... }
function handleSyncRoadmap(cwd, args, raw) { ... }
function handleSyncPhase(cwd, args, raw) { ... }
function handleUpdateStatus(cwd, args, raw) { ... }
function handleAddComment(cwd, args, raw) { ... }
function handleSyncStatus(cwd, args, raw) { ... }
function handlePrepareRoadmapSync(cwd, args, raw) { ... }
function handlePreparePhaseSync(cwd, args, raw) { ... }
function handleRecordMapping(cwd, args, raw) { ... }
function handleRecordStatus(cwd, args, raw) { ... }
function handleSchedule(cwd, args, raw) { ... }
function handlePrepareReschedule(cwd, args, raw) { ... }
```

Then replace cmdTracker with a clean dispatcher:

```js
function cmdTracker(cwd, subcommand, args, raw) {
  const handlers = {
    'get-config': handleGetConfig,
    'sync-roadmap': handleSyncRoadmap,
    'sync-phase': handleSyncPhase,
    'update-status': handleUpdateStatus,
    'add-comment': handleAddComment,
    'sync-status': handleSyncStatus,
    'prepare-roadmap-sync': handlePrepareRoadmapSync,
    'prepare-phase-sync': handlePreparePhaseSync,
    'record-mapping': handleRecordMapping,
    'record-status': handleRecordStatus,
    'schedule': handleSchedule,
    'prepare-reschedule': handlePrepareReschedule,
  };

  const handler = handlers[subcommand];
  if (!handler) {
    error(`Unknown tracker subcommand: '${subcommand}'. Available: ${Object.keys(handlers).join(', ')}`);
  }
  handler(cwd, args, raw);
}
```

The handler functions are NOT exported — they are internal to the module. Only `cmdTracker` remains exported.
