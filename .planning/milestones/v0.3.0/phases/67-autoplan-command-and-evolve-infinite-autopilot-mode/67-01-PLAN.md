    ```typescript
    /**
     * Options for the autoplan command.
     * Controls discovery behavior, subprocess parameters, and output format.
     */
    export interface AutoplanOptions {
      groups?: Array<{
        id: string;
        theme: string;
        dimension: string;
        items: Array<{ title: string; description: string; effort: string }>;
        priority: number;
        effort: string;
      }>;                       // Pre-discovered work groups (skip discovery if provided)
      pickPct?: number;         // Discovery pick percentage (only used when groups not provided)
      dryRun?: boolean;         // Build prompt only, do not spawn subprocess
      timeout?: number;         // Subprocess timeout in minutes
      maxTurns?: number;        // Max turns for claude -p subprocess
      model?: string;           // Model override for subprocess
      milestoneName?: string;   // Override for milestone name (default: derived from groups)
    }

    /**
     * Result returned by runAutoplan().
     */
    export interface AutoplanResult {
      status: 'completed' | 'failed' | 'dry-run';
      groups_count: number;      // Number of work groups used as input
      items_count: number;       // Total work items across groups
      prompt: string;            // The prompt that was (or would be) sent to claude -p
      milestone_name?: string;   // Derived or overridden milestone name
      reason?: string;           // Failure reason if status is 'failed'
    }
    ```

    Create a new module `lib/autoplan.ts` following the exact patterns established in `lib/autopilot.ts`:

    **Module header:**
    - `'use strict';` at top
    - JSDoc module description
    - `import type` for AutoplanOptions, AutoplanResult, GrdConfig, MilestoneInfo from ./types
    - `import type` for WorkGroup, GroupDiscoveryResult, EvolveGroupState, EvolveState from ./evolve/types

    **Imports (require-as pattern):**
    - `const { loadConfig, output, getMilestoneInfo } = require('./utils') as { ... }`
    - `const { spawnClaude } = require('./autopilot') as { ... }`
    - `const { runGroupDiscovery } = require('./evolve/discovery') as { ... }`
    - `const { readEvolveState } = require('./evolve/state') as { ... }`

    **Functions to implement:**

    1. `buildAutoplanPrompt(groups: WorkGroup[], milestoneName?: string): string`
       - Builds a prompt string for `claude -p` that instructs it to create a new milestone
       - The prompt must:
         a. Tell claude to invoke the `grd:new-milestone` skill in autonomous mode
         b. Include a structured summary of the discovered work groups as context: group themes, item titles, priorities, effort levels
         c. Instruct claude to use the work groups as the basis for milestone phases (each high-priority group = 1 phase, related groups can be combined)
         d. Specify the milestone name if provided
         e. Say "Autonomous mode — make all decisions yourself, no questions."
       - Format the groups as a markdown list inside the prompt string
       - Return the complete prompt string

    2. `runAutoplan(cwd: string, options: AutoplanOptions = {}): AutoplanResult`
       - If `options.groups` is provided and non-empty, use those directly
       - If `options.groups` is not provided, run discovery: call `runGroupDiscovery(cwd, readEvolveState(cwd), options.pickPct)`  to get selected_groups. Since runGroupDiscovery is async, runAutoplan must be async too.
       - If no groups found (either provided or discovered), return `{ status: 'failed', groups_count: 0, items_count: 0, prompt: '', reason: 'No work groups available for autoplan' }`
       - Derive `milestoneName` from groups if not provided in options: use the theme of the highest-priority group, or "Improvements" as fallback
       - Call `buildAutoplanPrompt(groups, milestoneName)` to build the prompt
       - If `options.dryRun`, return `{ status: 'dry-run', groups_count, items_count, prompt, milestone_name }`
       - Otherwise, call `spawnClaude(cwd, prompt, { timeout: timeoutMs, maxTurns, model })` -- use synchronous spawn since autoplan creates artifacts that must exist before returning
       - If exitCode !== 0, return failed result with reason
       - If exitCode === 0, return completed result
       - Use the same `timeoutMs = timeout ? timeout * 60 * 1000 : undefined` pattern from autopilot.ts

    3. `cmdAutoplan(cwd: string, args: string[], raw: boolean): Promise<void>`
       - Parse flags using the established flag/hasFlag pattern:
         - `--dry-run` (boolean)
         - `--timeout N` (number, minutes)
         - `--max-turns N` (number)
         - `--model <model>` (string)
         - `--pick-pct N` (number)
         - `--name <name>` (string, milestone name override)
       - Call `runAutoplan(cwd, options)`
       - Call `output(result, raw, rawSummary)` with a summary like "Autoplan: {status} ({groups_count} groups, {items_count} items)"

    4. `cmdInitAutoplan(cwd: string, raw: boolean): void`
       - Load config via `loadConfig(cwd)`
       - Read evolve state via `readEvolveState(cwd)`
       - Get milestone info via `getMilestoneInfo(cwd)`
       - Return a JSON object with:
         - `evolve_state`: { exists, iteration, remaining_groups_count, all_items_count }
         - `current_milestone`: { version, name }
         - `config`: { model_profile, autonomous_mode }
       - Call `output(result, raw)`

    **Module exports:**

    ```typescript
    };
    ```

    - All functions must have explicit return type annotations
